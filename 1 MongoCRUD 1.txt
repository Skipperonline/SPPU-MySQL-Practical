2. Find male students from “TE A” OR with grade ≥ 31
db.student.find({
  $or: [
    { $and: [ { gender: "male" }, { class: "TE A" } ] },
    { grd_point: { $gte: 31 } }
  ]
}).pretty()


3. Find students from Bangalore aged ≤ 19
db.student.find({ city: "Bangalore", age: { $lte: 19 } }).pretty()


4. Update course to “MCA” where first name = “Bhakti”
db.student.updateMany({ first_name: "Bhakti" }, { $set: { course: "MCA" } })
# verify
db.student.find({ first_name: "Bhakti" }).pretty()


5. Delete male students aged 18
db.student.deleteMany({ age: 18, gender: "male" })
# verify
db.student.find({ age: 18, gender: "male" }).count()  


6. Delete student where last name = “Jadhav” and class = “TE B”
db.student.deleteMany({ last_name: "Jadhav", class: "TE B" })
# verify
db.student.find({ last_name: "Jadhav", class: "TE B" }).count()
