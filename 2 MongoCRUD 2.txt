Step 4: Find documents based on conditions
gender = female OR (class = TE B AND grd_point ≥ 31)

db.student.find({
  $or: [
    { gender: "female" },
    { $and: [ { class: "TE B" }, { grd_point: { $gte: 31 } } ] }
  ]
}).pretty()


Step 5: Update age to 20 where fname = "Riyansh"

  { fname: "Riyansh" },

Step 6: Update or save city and state where lname = "Soniminde"
db.student.updateOne(
  { lname: "Soniminde" },
  { $set: { city: "Nagpur", state: "MH" } },

Step 7: Delete all students belonging to state “KA”

Step 8: Delete student where fname = "Evanshika" and age < 18
